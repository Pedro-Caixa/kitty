Queue

Queue class with better performance characteristics than table.remove()

local queue = Queue.new()
queue:PushRight("a")
queue:PushRight("b")
queue:PushRight("c")

while not queue:IsEmpty() do
    local entry = queue:PopLeft()
    print(entry) --> a, b, c
end
